274  /*
275  Problem: The string comparison algorithms compare character by character and skip any leading or trailing whitespace between terminals.
276  Rules 'ABC' + 'XYZ' maybe defined expecting the two terminals be separated by whitespace but this algorithm could parse the string 'ABCXYZ' as two separate terminals.
277  There's a number of traditional approaches to solving this such as tokenizing before parsing or creating parse tables.
278  A proper handling would compound the complexity of this library beyond it's intended scope, there are plenty of complex parsers around.
279  Currently, the last character parsed is checked against the next character in the stream to see if they're of the same 'class' and fail if so.
280  This isn't ideal because it maybe perfectly valid in some grammars to expect 'ABCXYZ' to appear in the input stream yet successfully parse into independent terminals.
281  This library assumes contiguous alpha-numeric terminals constitute a single terminal so the input stream of 'ABCXYZ' will fail to parse without grammar definition trickery
282  */
283  if (oCurr < end && isalnum(*oCurr) && isalnum(_str[_len - 2])){
284  return rule_base::pointer_type(nullptr);
285  }
286  parse_helper< _WhitespaceT, void, true, void>::parse(oCurr, end);
287 
288  begin = oCurr;
289  return rule_base::pointer_type(new _DeclT);
290  }
